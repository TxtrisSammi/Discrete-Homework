\question Give examples of three sets $A, B$ and $C$ such that
\begin{enumerate}[label=(\alph*)]
    \item $A \subseteq B \not\subset C$.

    % solution goes here
    \begin{sol}
        $A=\{1,2,3\}$
        $B=\{1,2,3\}$
        $C=\{0\}$
    \end{sol}

    
    \item $A \subseteq B, B \in C$ and $A \cap C = \nil$.

    % solution goes here
    \begin{sol}
        $A=\{1\}$
        $B=\{1,2,3\}$
        $C=\{\{1,2,3\},3\}$
    \end{sol}

    % solution goes here

        $C=\{\{\{3\}\}\}$
    \end{sol}
\end{enumerate}
